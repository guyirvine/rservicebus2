def getResources( env )
	resources = Hash.new

	env.each do |k,v|
		if v.is_a?(String) and
				k.start_with?( "RSB_" ) then
			uri = URI.parse( v )
			case uri.scheme
				when "redis"
					resources[k.sub( "RSB_", "" )] = RedisAppResource.new( uri )
				else
					abort("Scheme, #{uri.scheme}, not recognised when configuring app resource, #{k}=#{v}");
			end
		end
				
	end

	return resources
end
